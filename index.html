} function importPDFtoSheet() { const FOLDER_ID = '1HC4miyMDWYb9C82fnBKuLOMR3Rxkj1fy'; const SHEET_NAME = '実績データ'; const folder = DriveApp.getFolderById(FOLDER_ID); const files = folder.getFilesByType(MimeType.PDF); if (/^-?\d+$/.test(cleanT)) nums.push(parseInt(cleanT, 10)); if (nums.length === 2) break; } if (nums.length === 2) matches.push({ region: tokens[i] === '計' ?
